Usage Code

<script src="https://cdn.jsdelivr.net/npm/@coolbuttons/vanilla@1.0.4/dist/coolbuttons.min.js"></script>

<script>
  const { CoolButton } = window.CoolButtonsVanilla;
  
  const button = new CoolButton({
    text: 'Click Me!',
    className: 'px-4 py-2 bg-orange-500 text-white rounded hover:bg-orange-600',
    onClick: () => alert('Button clicked!')
  });
  
  document.getElementById('app').appendChild(button.getElement());
</script>
